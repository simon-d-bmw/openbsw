40  template<typename Tuple, uint32_t I>
41  class TryDo
42  {
43  Tuple& _tuplePools;
44 
45  public:
46  explicit TryDo(Tuple& pools) : _tuplePools(pools) {}
47 
48  uint8_t* allocate(uint32_t const size, PoolStats* poolStats = nullptr)
49  {
50  auto& currentPool = etl::get<I>(_tuplePools);
51  uint8_t* const ptr = currentPool.allocate(size);
52  if (ptr != nullptr)
53  {
54  if (poolStats != nullptr)
55  {
56  poolStats->delegatedAllocations++;
57  }
58  return ptr;
59  }
60  if ((size <= currentPool.chunkSize()) && (poolStats == nullptr) && currentPool.isFull())
61  {
62  poolStats = &(currentPool.getPoolStats());
63  }
64  return TryDo<Tuple, I + 1U>(_tuplePools).allocate(size, poolStats);
65  }
66 
67  void deallocate(void* const ptr)
68  {
69  auto& pool = etl::get<I>(_tuplePools);
70  if (pool.deallocate(ptr))
71  {
72  return;
73  }
74  TryDo<Tuple, I + 1U>(_tuplePools).deallocate(ptr);
75  }
76 
77  bool isPtrValid(void const* const ptr)
78  {
79  auto& pool = etl::get<I>(_tuplePools);
80  uint8_t const* const bufferPtr = reinterpret_cast<uint8_t const*>(ptr); // NOLINT
81  if (pool.isValidPointer(bufferPtr))
82  {
83  return true;
84  }
85  return TryDo<Tuple, I + 1U>(_tuplePools).isPtrValid(ptr);
86  }
87 
88  void collectStats(etl::delegate<void(size_t const, PoolStats const)> const& collector)
89  {
90  auto& pool = etl::get<I>(_tuplePools);
91  PoolStats const stats = pool.getPoolStats();
92  collector(I, stats);
93  pool.resetStats();
94  TryDo<Tuple, I + 1U>(_tuplePools).collectStats(collector);
95  }
96 
97  void initialize()
98  {
99  auto& pool = etl::get<I>(_tuplePools);
100  pool.initialize();
101  TryDo<Tuple, I + 1U>(_tuplePools).initialize();
102  }
