56  static void call(Callback const& callback, Message const& msg, Future::State const state)
57  {
58  if (state != Future::State::Ready)
59  {
60  callback.call_if(etl::unexpected<Future::State>(etl::in_place_t{}, state));
61  }
62  else
63  {
64  callback.call_if(etl::expected<ArgumentType, Future::State>(
65  etl::in_place_t{},
