301  etl::expected<uint16_t, HRESULT>
302  set(SetterArgumentType const& payload, SetterCallback const& cbk)
303  {
304  etl::expected<uint16_t, HRESULT> result

306 
307  if (Base::_proxy->isInitialized())
308  {
309  etl::optional<uint16_t> reqId = _setDispatcher.obtainRequestId(cbk);
310  if (reqId.has_value())
311  {
312  Message msg = Base::_proxy->generateMessageHeader(
313  SetterTraits::METHOD_MEMBER_ID, reqId.value());
314  HRESULT res = MessagePayloadBuilder::getInstance().allocate(payload, msg);
315  if (res == HRESULT::Ok)
316  {
317  res = Base::_proxy->sendMessage(msg);
318  if (res == HRESULT::Ok)
319  {
320  result = reqId.value();
321  }
322  else
323  {
324  MessagePayloadBuilder::deallocate(msg);
325  logger::logMessageSendingFailure(
326  logger::LogLevel::Error, logger::Error::SendMessage, res, msg);
327  result = etl::unexpected<HRESULT>(etl::in_place_t{}, res);
328  }
329  }
330  else
331  {
332  result = etl::unexpected<HRESULT>(etl::in_place_t{}, res);
333  }
334 
335  if (!result.has_value())

338  }
339  }
